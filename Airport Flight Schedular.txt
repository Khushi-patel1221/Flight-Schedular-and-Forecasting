void sortflight()
{
    for(int i=0;i<v.size();i++)
    {
        for(int j=i+1;j<v.size();j++)
        {
            if(stoi(v[i][2])>stoi(v[j][2]))
            swap(v[i],v[j]);
        }
    }
}

void scheduleflight()
{
    int sumrunway=0,sumc=0;
    vector<vector<int>> r(runway);
    int x=0;
    queue <int> ontime;
    for(int i=0;i<60;i++)
    {
        map <int,int> ma,md;
        for(int j=0;j<v.size();j++)
        {
            int xy=stoi(v[j][2]);
            if(xy==i)
            {x++;
            int y=stoi(v[j][4]);
            if(v[j][3]=="A")
            ma[y]=j+1;
            else
            md[y]=j+1;
            }
        }
        for(auto y:ma)
        ontime.push(y.second);
        for(auto y:md)
        ontime.push(y.second);
        if(x>runway)
        {
            for(int z=0;z<runway;z++)
            {
                if(!ontime.empty())
                {
                    r[z].push_back(ontime.front());
                    ontime.pop();
                    sumrunway++;
                }
                else
                r[z].push_back(0);
            }
            x=x-runway;
            sumc=sumc+x;
        }
        else
        {
            
            for(int z=0;z<runway;z++)
            {
                if(!ontime.empty())
                {
                    r[z].push_back(ontime.front());
                    ontime.pop();
                    sumrunway++;
                }
                else
                r[z].push_back(0);
            }
            x=0;
        }
    }
    for(int i=0;i<r.size();i++)
    {
        cout<<"R"<<i+1<<" ";
        for(int j=0;j<r[i].size();j++)
        {
            cout<<r[i][j]<<" ";
        }
        cout<<endl;
    }
    cout<<"Flights that got congested : "<<sumc<<endl;
    cout<<"Average runway usage is "<<sumrunway/60<<endl;
    cout<<"Efficiency of runway is "<< (double)v.size()/(runway*60)<<endl;
    cout<<"Number of Empty Windows for New flights"<< 60*runway-sumrunway<<endl;
}
